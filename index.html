SDK / Code-Beispiele

Integration in 5–10 Zeilen.

JavaScript

import { DeuroClient } from "@deuro/sdk";

const client = new DeuroClient({ baseUrl: "https://api.example.dev", apiKey: "SANDBOX_KEY" });
await client.sendPayment({ from: "DEAN123", to: "DEAN456", amount: 10.00 });
console.log("Payment sent");

Go

import (
  "context"
  deuro "github.com/you/deuro-sdk-go"
)

cli := deuro.New("https://api.example.dev", "SANDBOX_KEY")
_ = cli.SendPayment(context.Background(), deuro.Payment{From:"DEAN123", To:"DEAN456", Amount:10})
